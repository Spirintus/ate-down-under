liberator_invasion = { #aka subjugation invasion hybrid
	name = CB_NAME_LIBERATION
	war_name = WAR_NAME_LIBERATION
	sprite = 8
	truce_days = 1
	is_permanent = no
	can_ask_to_join_war = no
	allowed_to_target_tributaries = no
	defender_unoccupied_warscore = yes
	allow_whitepeace = no
	
	can_use = {
		always = no
	}
	
	is_valid = {
		ROOT = { has_character_flag = this_is_liberator}
	}
	
	on_success = {
		ROOT = {
			piety = 100
			participation_scaled_prestige = 100
		}
		# any_attacker = {
			# limit = { NOT = { character = ROOT } }
			# hidden_tooltip = { participation_scaled_prestige = 100 }
		# }
		ROOT = {
			occupy_minors_of_occupied_settlements = FROM
			gain_all_occupied_titles = FROM
			
			if = {
				limit = {
					has_nickname = no
				}
				give_nickname = nick_the_liberator
			}
			
			hidden_tooltip = {
				if = { #if liberator isn't king of dunedin then he has just conquered Dunedin and therefore he is crowned
					limit = {
						has_character_flag = this_is_liberator
						not = { k_dunedin = { holder_scope = { character = ROOT } } }
					}
					narrative_event = {id = erulib.3 days = 7}
				}
			}
		}
		FROM = {
			piety = -100
			prestige = -100
			hidden_tooltip = {
				opinion = {
					modifier = opinion_subjugated 
					who = ROOT
				}
				any_vassal = {
					opinion = {
						modifier = opinion_subjugated 
						who = ROOT
					}
				}				
				any_demesne_title = {
					limit = {
						higher_tier_than = ROOT
					}
					ROOT = { usurp_title = { target = PREV type = invasion } }
					# add_pressed_claim = PREV
				}
				any_demesne_title = {
					limit = {
						tier = ROOT
					}
					ROOT = { usurp_title = { target = PREV type = invasion } }
					# add_pressed_claim = PREV
				}
				set_defacto_liege = ROOT
			}
		}
	}
	
	on_fail = {
		FROM = {
			piety = 50
			participation_scaled_prestige = 50
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 50 }
		}
		ROOT = {
			piety = -100
		}
	}
	
	on_invalidation = {
		ROOT = {
			any_demesne_title = {
				limit = {
					OR = {
						tier = king
						tier = emperor
						title = d_liberator
					}
				}
				destroy_landed_title = THIS
			}
		}
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 2.0
			}
			death = { 
				death_reason = death_execution 
				killer = FROM 
			}			
		}
		FROM = {
			piety = 100
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		ROOT = {
			any_demesne_title = {
				limit = {
					OR = {
						tier = king
						tier = emperor
						title = d_liberator
					}
				}
				destroy_landed_title = THIS
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
